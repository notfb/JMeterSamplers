265 	/**
266 	 * @see org.apache.jmeter.samplers.Sampler#sample(org.apache.jmeter.samplers.Entry)
267 	 */
268 	public SampleResult sample(final Entry entry) {
269 		final SampleResult sresult = new SampleResult();
270 		sresult.setSuccessful(false);
271 		sresult.setDataType(SampleResult.TEXT);
272 
273 		if (getClassname() == null) {
274 			sresult.setResponseMessage("Can not sample - no classname set!");
275 			return sresult;
276 		}
277 
278 		final Result tr;
279 
280 		sresult.setSampleLabel(getName());
281 		sresult.setSamplerData(getClassname() + "." + getMethod());
282 
283 		// TODO: optimize
284 		final Object testInstance = getClassInstance(getClassname());
285 		if (testInstance == null) {
286 			sresult.setResponseCode("failed to init class instance, check logfile ...");
287 			return sresult;
288 		}
289 
290 		try {
291 			// final TestSuite ts = new TestSuite();
292 			// ts.runTest( testCase, tr);
293 			// TODO: run just one test method ...
294 			// TODO: use junit timing?
295 			sresult.sampleStart();
296 			tr = JUnitCore.runClasses(testInstance.getClass());
297 		} catch (final RuntimeException e) {
298 			log.warn("exectution of junit test failed", e);
299 			sresult.setResponseCode(getErrorCode());
300 			sresult.setResponseMessage(getError());
301 			if (getAppendException()) {
302 				sresult.setResponseData(String.format("%s - %s \n%s",
303 						e.getClass().getName(), e.getMessage(),
304 						Arrays.toString(e.getStackTrace())).getBytes());
305 			}
306 			return sresult;
307 		} finally {
308 			sresult.sampleEnd();
309 		}
310 
311 		if (tr == null) {
312 			log.error("junit4 test result was null!");
313 			sresult.setResponseCode("-1"); // TODO:
314 			sresult.setResponseMessage("test result was null");
315 		} else if (tr.wasSuccessful()) {
316 			log.debug(String.format(
317 					"junit4 test successfull! time: %sms testcount: %s",
318 					tr.getRunTime(), tr.getRunCount()));
319 			sresult.setSuccessful(true);
320 			sresult.setResponseMessage(getSuccess());
321 			sresult.setResponseCode(getSuccessCode());
322 			sresult.setResponseData(String.format("%s testcount: %s",
323 					getSuccess(), tr.getRunCount()).getBytes());
324 		} else {
325 			// test failed, append failure traces etc.
326 			log.debug(String.format(
327 					"junit4 test failed! time: %sms testcount: %s failures %s",
328 					tr.getRunTime(), tr.getRunCount(), tr.getFailureCount()));
329 
330 			final StringBuffer buf = new StringBuffer();
331 			buf.append(getFailure());
332 			buf.append(" testcount: ").append(tr.getRunCount());
333 			buf.append(" failures: ").append(tr.getFailureCount());
334 			buf.append("\n");
335 			if (getAppendError()) {
336 				final List<Failure> failures = tr.getFailures();
337 				for (final Failure f : failures) {
338 					buf.append("\nFailure: msg: ").append(f.getMessage());
339 					buf.append(" descr.: ").append(f.getDescription());
340 					buf.append(" header: ").append(f.getTestHeader());
341 					if (getAppendException()) {
342 						buf.append("\nstacktrace: ");
343 						buf.append(f.getTrace());
344 						buf.append("\n");
345 					}
346 				}
347 			}
348 
349 			sresult.setResponseData(buf.toString().getBytes());
350 			sresult.setResponseMessage(getFailure());
351 			sresult.setResponseCode(getFailureCode());
352 		}
353 
354 		return sresult;
355 	}

357 	/**
358 	 * 
359 	 * @param clazz
360 	 * @param method
361 	 * @return the method or null if an error occurred
362 	 */
363 	@SuppressWarnings("unused")
364 	private Method getMethod(final Class<?> clazz, final String method) {
365 		if (clazz != null && method != null) {
366 			// log.info("class " + clazz.getClass().getName() +
367 			// " method name is " + method);
368 			try {
369 				return clazz.getMethod(method, new Class[0]);
370 			} catch (final NoSuchMethodException e) {
371 				log.warn(e.getMessage());
372 			}
373 		}
374 		return null;
375 	}

377 	public static Object getClassInstance(final String className) {
378 
379 		if (className == null || "".equals(className.trim())) {
380 			log.info("getClassInstance called with empty / null string - ignoreing ...");
381 			return null;
382 		}
383 
384 		try {
385 			final Class<?> clazz = Class.forName(className);
386 			return clazz.getConstructor(new Class[0]).newInstance(new Object[0]);
387 		} catch (final RuntimeException e) {
388 			log.warn("failed to create class instance ", e);
389 		} catch (final ClassNotFoundException e) {
390 			log.warn("class not found ", e);
391 		} catch (final NoSuchMethodException e) {
392 			log.warn("constuctor not found ", e);
393 		} catch (final InstantiationException e) {
394 			log.warn("constructor call failed ", e);
395 		} catch (final IllegalAccessException e) {
396 			log.warn("constructor call failed ", e);
397 		} catch (final InvocationTargetException e) {
398 			log.warn("constructor call failed ", e);
399 		}
400 
401 		// caller handles null
402 		return null;
403 	}
